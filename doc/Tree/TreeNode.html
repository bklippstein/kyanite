def find( node_name )
  return self               if self.name == node_name
  result = self[node_name]
  return result if result
  children do |c| 
    next if c.children.empty?
    #puts "durchsuche #{c.name}"
    result = c.find(node_name)
    return result if result
  end
  nil
end

    print "*"
